            \textbf{Score Function for Biclustering}
            \begin{itemize}
                \item $$\langle x, y \rangle = \exp(-\frac{|x - y|^2}{2\alpha|x||y|})$$
                \item Use the inner product above to substitute for the co-variance
            \end{itemize}
        \end{column}

            \textbf{Biclustering}
